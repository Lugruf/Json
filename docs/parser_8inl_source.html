294  size_t count_array_elements(TokenForwardIterator begin, TokenForwardIterator end)

326  if (square_bracket_level == 0)
327  break;

330  }
331

391  template <typename CharForwardIterator>
392  json::value parse_string(const token<CharForwardIterator> & t)
393  {
394  // Remove the first and last characters as the "" are not part of the string
395  json::string str{ std::next(t.begin), t.end };
396  str.pop_back(); // Use pop_back instead std::prev(t.end) in order to allow forward iterators
397 
398  // Scan for '\\' to look for control characters
399  for (auto it = str.begin(); it != str.end(); ++it)
400  {
401  if (*it == '\\')
402  {
403  // Avoid dereferencing the end iterator
404  const char next = it + 1 == str.end() ? '\0' : it[1];
405  switch (next)

408  case '"':
409  case '\\':

419  }

422  it = str.erase(it);
423  *it = '\f';
424  break;
425  }
426  case 'n':
427  {
428  it = str.erase(it);
429  *it = '\n';
