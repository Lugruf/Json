505  json::value_type value::type() const noexcept
506  {
507  // stored_value.index() will return
508  // null = -1
509  // integer = 0
510  // real = 1
511  // boolean = 2
512  // string = 3
513  // array = 4
514  // object = 5
515  // We have to map this to the enum value_type, where
516  // null = 0
517  // integer = 1
518  // real = 2
519  // boolean = 3
520  // string = 4
521  // array = 5
522  // object = 6
523  // So just add 1 to the return value of index() to get the current type in the enum
524 
525  return static_cast<json::value_type>(stored_value.index() + 1);
